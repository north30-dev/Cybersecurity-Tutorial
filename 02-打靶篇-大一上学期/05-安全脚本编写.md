#!/usr/bin/env python3
"""
子域名枚举脚本
支持多种枚举方式
"""

import requests
import dns.resolver
import concurrent.futures
from typing import List, Set
import argparse

class SubdomainEnumerator:
    def __init__(self, domain: str, threads: int = 20):
        self.domain = domain
        self.threads = threads
        self.found: Set[str] = set()
    
    def certificate_transparency(self) -> List[str]:
        """证书透明度查询"""
        url = f"https://crt.sh/?q=%25.{self.domain}&output=json"
        try:
            resp = requests.get(url, timeout=30)
            if resp.status_code == 200:
                data = resp.json()
                subdomains = []
                for entry in data:
                    name = entry.get('name_value', '')
                    # 处理通配符
                    name = name.replace('*.', '')
                    if name.endswith(self.domain):
                        subdomains.append(name)
                return list(set(subdomains))
        except Exception as e:
            print(f"[-] CT查询失败: {e}")
        return []
    
    def dns_resolve(self, subdomain: str) -> bool:
        """DNS解析验证"""
        try:
            resolver = dns.resolver.Resolver()
            resolver.timeout = 3
            resolver.lifetime = 3
            answers = resolver.resolve(subdomain, 'A')
            return len(answers) > 0
        except:
            return False
    
    def brute_force(self, wordlist: str) -> List[str]:
        """字典爆破"""
        found = []
        
        try:
            with open(wordlist, 'r') as f:
                words = [line.strip() for line in f if line.strip()]
        except FileNotFoundError:
            print(f"[-] 字典文件不存在: {wordlist}")
            return []
        
        def check_subdomain(word: str) -> Optional[str]:
            subdomain = f"{word}.{self.domain}"
            if self.dns_resolve(subdomain):
                return subdomain
            return None
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(check_subdomain, word): word for word in words}
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                if result:
                    found.append(result)
                    print(f"[+] 发现: {result}")
        
        return found
    
    def run(self, wordlist: str = None) -> Set[str]:
        """执行完整枚举"""
        print(f"[*] 开始枚举子域名: {self.domain}")
        
        # 证书透明度
        print("[*] 查询证书透明度...")
        ct_results = self.certificate_transparency()
        self.found.update(ct_results)
        print(f"[+] CT发现 {len(ct_results)} 个子域名")
        
        # 字典爆破
        if wordlist:
            print("[*] 开始字典爆破...")
            brute_results = self.brute_force(wordlist)
            self.found.update(brute_results)
            print(f"[+] 爆破发现 {len(brute_results)} 个子域名")
        
        return self.found

def main():
    parser = argparse.ArgumentParser(description='子域名枚举工具')
    parser.add_argument('-d', '--domain', required=True, help='目标域名')
    parser.add_argument('-w', '--wordlist', help='字典文件')
    parser.add_argument('-o', '--output', help='输出文件')
    parser.add_argument('-t', '--threads', type=int, default=20, help='线程数')
    args = parser.parse_args()
    
    enumerator = SubdomainEnumerator(args.domain, args.threads)
    results = enumerator.run(args.wordlist)
    
    print(f"\n[*] 共发现 {len(results)} 个子域名")
    
    if args.output:
        with open(args.output, 'w') as f:
            for sub in sorted(results):
                f.write(f"{sub}\n")
        print(f"[+] 结果已保存到: {args.output}")

if __name__ == '__main__':
    main()

#!/usr/bin/env python3
"""
端口扫描脚本
支持TCP/UDP扫描
"""

import socket
import concurrent.futures
from typing import List, Dict, Tuple
import argparse
from dataclasses import dataclass

@dataclass
class PortResult:
    port: int
    status: str
    service: str = ""

class PortScanner:
    # 常见端口服务映射
    COMMON_PORTS = {
        21: 'FTP', 22: 'SSH', 23: 'Telnet', 25: 'SMTP',
        53: 'DNS', 80: 'HTTP', 110: 'POP3', 143: 'IMAP',
        443: 'HTTPS', 445: 'SMB', 993: 'IMAPS', 995: 'POP3S',
        1433: 'MSSQL', 1521: 'Oracle', 3306: 'MySQL',
        3389: 'RDP', 5432: 'PostgreSQL', 5900: 'VNC',
        6379: 'Redis', 8080: 'HTTP-Alt', 8443: 'HTTPS-Alt',
        27017: 'MongoDB'
    }
    
    def __init__(self, target: str, threads: int = 100):
        self.target = target
        self.threads = threads
        self.results: List[PortResult] = []
    
    def scan_port(self, port: int) -> PortResult:
        """扫描单个端口"""
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(2)
            result = sock.connect_ex((self.target, port))
            sock.close()
            
            if result == 0:
                service = self.COMMON_PORTS.get(port, 'unknown')
                return PortResult(port, 'open', service)
            return PortResult(port, 'closed')
        except:
            return PortResult(port, 'filtered')
    
    def scan_range(self, start: int, end: int) -> List[PortResult]:
        """扫描端口范围"""
        ports = range(start, end + 1)
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(self.scan_port, port): port for port in ports}
            
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                if result.status == 'open':
                    self.results.append(result)
                    print(f"[+] {result.port}/tcp\t{result.status}\t{result.service}")
        
        return self.results
    
    def scan_common(self) -> List[PortResult]:
        """扫描常见端口"""
        ports = list(self.COMMON_PORTS.keys())
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(self.scan_port, port): port for port in ports}
            
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                if result.status == 'open':
                    self.results.append(result)
                    print(f"[+] {result.port}/tcp\t{result.status}\t{result.service}")
        
        return self.results
    
    def banner_grab(self, port: int) -> str:
        """获取服务Banner"""
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(5)
            sock.connect((self.target, port))
            sock.send(b'HEAD / HTTP/1.0\r\n\r\n')
            banner = sock.recv(1024).decode().strip()
            sock.close()
            return banner
        except:
            return ""

def main():
    parser = argparse.ArgumentParser(description='端口扫描工具')
    parser.add_argument('-t', '--target', required=True, help='目标IP')
    parser.add_argument('-p', '--ports', default='1-1000', help='端口范围(如: 1-1000)')
    parser.add_argument('--common', action='store_true', help='仅扫描常见端口')
    parser.add_argument('-o', '--output', help='输出文件')
    args = parser.parse_args()
    
    scanner = PortScanner(args.target)
    
    if args.common:
        print(f"[*] 扫描常见端口: {args.target}")
        results = scanner.scan_common()
    else:
        start, end = map(int, args.ports.split('-'))
        print(f"[*] 扫描端口 {start}-{end}: {args.target}")
        results = scanner.scan_range(start, end)
    
    print(f"\n[*] 发现 {len(results)} 个开放端口")
    
    if args.output:
        with open(args.output, 'w') as f:
            for r in results:
                f.write(f"{r.port},{r.status},{r.service}\n")

if __name__ == '__main__':
    main()

```python
#!/usr/bin/env python3
"""
目录枚举脚本
支持多线程扫描
"""

import requests
import concurrent.futures
from typing import List, Dict
import argparse
from urllib.parse import urljoin

class DirectoryEnumerator:
    def __init__(self, base_url: str, threads: int = 20):
        self.base_url = base_url.rstrip('/')
        self.threads = threads
        self.results: List[Dict] = []
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        })
    
    def check_path(self, path: str) -> Dict:
        """检查单个路径"""
        url = urljoin(self.base_url + '/', path)
        try:
            resp = self.session.get(url, timeout=5, allow_redirects=False)
            
            if resp.status_code in [200, 301, 302, 403]:
                return {
                    'url': url,
                    'status': resp.status_code,
                    'length': len(resp.content),
                    'redirect': resp.headers.get('Location', '')
                }
        except requests.RequestException:
            pass
        return {}
    
    def scan(self, wordlist: str, extensions: List[str] = None) -> List[Dict]:
        """执行扫描"""
        try:
            with open(wordlist, 'r') as f:
                paths = [line.strip() for line in f if line.strip() and not line.startswith('#')]
        except FileNotFoundError:
            print(f"[-] 字典文件不存在: {wordlist}")
            return []
        
        # 添加扩展名
        if extensions:
            extended_paths = []
            for path in paths:
                extended_paths.append(path)
                for ext in extensions:
                    extended_paths.append(f"{path}{ext}")
            paths = extended_paths
        
        print(f"[*] 开始扫描: {self.base_url}")
        print(f"[*] 总计 {len(paths)} 个路径")
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(self.check_path, path): path for path in paths}
            
            for future in concurrent.futures.as_completed(futures):
                result = future.result()
                if result:
                    self.results.append(result)
                    status = result['status']
                    url = result['url']
                    length = result['length']
                    print(f"[{status}] {url} ({length} bytes)")
        
        return self.results
    
    def save_results(self, filename: str):
        """保存结果"""
        import json
        with open(filename, 'w') as f:
            json.dump(self.results, f, indent=2)

def main():
    parser = argparse.ArgumentParser(description='目录枚举工具')
    parser.add_argument('-u', '--url', required=True, help='目标URL')
    parser.add_argument('-w', '--wordlist', required=True, help='字典文件')
    parser.add_argument('-e', '--extensions', help='扩展名(如: php,html,js)')
    parser.add_argument('-t', '--threads', type=int, default=20, help='线程数')
    parser.add_argument('-o', '--output', help='输出文件')
    args = parser.parse_args()
    
    extensions = args.extensions.split(',') if args.extensions else None
    
    enumerator = DirectoryEnumerator(args.url, args.threads)
    results = enumerator.scan(args.wordlist, extensions)
    
    print(f"\n[*] 发现 {len(results)} 个有效路径")
    
    if args.output:
        enumerator.save_results(args.output)

if __name__ == '__main__':
    main()
```

#!/usr/bin/env python3
"""
Web指纹识别脚本
识别CMS、框架、服务器等
"""

import requests
import re
from typing import Dict, List
import argparse

class FingerprintScanner:
    def __init__(self, url: str):
        self.url = url.rstrip('/')
        self.session = requests.Session()
        self.fingerprints: Dict[str, str] = {}
    
    # CMS指纹库
    CMS_FINGERPRINTS = {
        'WordPress': [
            r'<meta name="generator" content="WordPress',
            r'/wp-content/',
            r'/wp-includes/',
            r'wp-emoji-release.min.js',
        ],
        'Drupal': [
            r'<meta name="generator" content="Drupal',
            r'/misc/drupal.js',
            r'Drupal\.settings',
        ],
        'Joomla': [
            r'<meta name="generator" content="Joomla',
            r'/media/jui/',
            r'/components/com_',
        ],
        'ThinkPHP': [
            r'ThinkPHP',
            r'think_show_page_trace',
        ],
        'Laravel': [
            r'laravel_session',
            r'XSRF-TOKEN',
        ],
        'Django': [
            r'csrfmiddlewaretoken',
            r'__admin_media_prefix__',
        ],
        'Spring': [
            r'Whitelabel Error Page',
            r'springframework',
        ],
    }
    
    # 服务器指纹
    SERVER_FINGERPRINTS = {
        'Apache': r'Apache[/\s]([\d.]+)?',
        'Nginx': r'nginx[/\s]([\d.]+)?',
        'IIS': r'Microsoft-IIS[/\s]([\d.]+)?',
        'Tomcat': r'Apache-Coyote[/\s]([\d.]+)?',
        'Jetty': r'Jetty[/\s]([\d.]+)?',
    }
    
    # 语言指纹
    LANGUAGE_FINGERPRINTS = {
        'PHP': [r'\.php', r'X-Powered-By:\s*PHP'],
        'Java': [r'\.jsp', r'\.do', r'JSESSIONID'],
        'Python': [r'\.py', r'uwsgi', r'gunicorn'],
        'Ruby': [r'\.rb', r'Rails', r'Rack'],
        'Node.js': [r'Express', r'connect\.sid'],
    }
    
    def fetch_page(self) -> tuple:
        """获取页面内容"""
        try:
            resp = self.session.get(self.url, timeout=10)
            return resp.text, resp.headers
        except Exception as e:
            print(f"[-] 获取页面失败: {e}")
            return "", {}
    
    def detect_cms(self, content: str) -> List[str]:
        """检测CMS"""
        detected = []
        for cms, patterns in self.CMS_FINGERPRINTS.items():
            for pattern in patterns:
                if re.search(pattern, content, re.IGNORECASE):
                    detected.append(cms)
                    break
        return detected
    
    def detect_server(self, headers: Dict) -> str:
        """检测服务器"""
        server_header = headers.get('Server', '')
        for server, pattern in self.SERVER_FINGERPRINTS.items():
            match = re.search(pattern, server_header, re.IGNORECASE)
            if match:
                version = match.group(1) if match.groups() else ''
                return f"{server} {version}".strip()
        return server_header or 'Unknown'
    
    def detect_language(self, content: str, headers: Dict) -> List[str]:
        """检测编程语言"""
        detected = []
        headers_str = str(headers)
        
        for lang, patterns in self.LANGUAGE_FINGERPRINTS.items():
            for pattern in patterns:
                if re.search(pattern, content + headers_str, re.IGNORECASE):
                    detected.append(lang)
                    break
        return detected
    
    def detect_framework(self, content: str, headers: Dict) -> List[str]:
        """检测框架"""
        frameworks = []
        
        # 检查常见框架特征
        framework_patterns = {
            'Vue.js': r'vue\.js|Vue\.',
            'React': r'react\.js|React|__REACT',
            'Angular': r'angular\.js|ng-',
            'jQuery': r'jquery\.js|jQuery',
            'Bootstrap': r'bootstrap\.css|Bootstrap',
        }
        
        for framework, pattern in framework_patterns.items():
            if re.search(pattern, content, re.IGNORECASE):
                frameworks.append(framework)
        
        return frameworks
    
    def scan(self) -> Dict:
        """执行完整扫描"""
        print(f"[*] 开始指纹识别: {self.url}")
        
        content, headers = self.fetch_page()
        
        # CMS检测
        cms = self.detect_cms(content)
        if cms:
            self.fingerprints['CMS'] = cms
            print(f"[+] CMS: {', '.join(cms)}")
        
        # 服务器检测
        server = self.detect_server(headers)
        self.fingerprints['Server'] = server
        print(f"[+] Server: {server}")
        
        # 语言检测
        languages = self.detect_language(content, headers)
        if languages:
            self.fingerprints['Language'] = languages
            print(f"[+] Language: {', '.join(languages)}")
        
        # 框架检测
        frameworks = self.detect_framework(content, headers)
        if frameworks:
            self.fingerprints['Framework'] = frameworks
            print(f"[+] Frontend Framework: {', '.join(frameworks)}")
        
        # 其他信息
        if 'X-Powered-By' in headers:
            self.fingerprints['Powered-By'] = headers['X-Powered-By']
            print(f"[+] Powered-By: {headers['X-Powered-By']}")
        
        return self.fingerprints

def main():
    parser = argparse.ArgumentParser(description='Web指纹识别工具')
    parser.add_argument('-u', '--url', required=True, help='目标URL')
    args = parser.parse_args()
    
    scanner = FingerprintScanner(args.url)
    results = scanner.scan()
    
    print("\n[*] 指纹识别结果:")
    for key, value in results.items():
        if isinstance(value, list):
            print(f"    {key}: {', '.join(value)}")
        else:
            print(f"    {key}: {value}")

if __name__ == '__main__':
    main()

#!/usr/bin/env python3
"""
SQL注入检测脚本
支持多种注入类型检测
"""

import requests
import re
from typing import Dict, List, Optional
import argparse
from urllib.parse import urlparse, parse_qs, urlencode, urlunparse

class SQLInjectionScanner:
    # SQL注入Payload
    PAYLOADS = {
        'error_based': [
            "'",
            '"',
            "' OR '1'='1",
            '" OR "1"="1',
            "' OR 1=1--",
            '" OR 1=1--',
            "1' AND '1'='1",
            "1' AND '1'='2",
        ],
        'union_based': [
            "' UNION SELECT NULL--",
            "' UNION SELECT NULL,NULL--",
            "' UNION SELECT NULL,NULL,NULL--",
            "' ORDER BY 1--",
            "' ORDER BY 10--",
        ],
        'time_based': [
            "' AND SLEEP(5)--",
            "' AND BENCHMARK(5000000,SHA1('test'))--",
            "'; WAITFOR DELAY '0:0:5'--",
        ],
        'boolean_based': [
            "' AND '1'='1",
            "' AND '1'='2",
            "' AND 1=1--",
            "' AND 1=2--",
        ]
    }
    
    # 错误特征
    ERROR_PATTERNS = [
        r'SQL syntax.*MySQL',
        r'Warning.*mysql_',
        r'MySqlException',
        r'PostgreSQL.*ERROR',
        r'Warning.*pg_',
        r'ORA-\d{5}',
        r'Microsoft SQL Server',
        r'SQLite3::SQLException',
        r'SQLite/JDBCDriver',
        r'SQL syntax error',
        r'unterminated quoted string',
    ]
    
    def __init__(self, url: str):
        self.url = url
        self.session = requests.Session()
        self.vulnerabilities: List[Dict] = []
    
    def parse_url(self) -> tuple:
        """解析URL"""
        parsed = urlparse(self.url)
        params = parse_qs(parsed.query)
        return parsed, params
    
    def test_parameter(self, param: str, original_value: str) -> Optional[Dict]:
        """测试单个参数"""
        parsed, params = self.parse_url()
        
        # 获取原始响应
        original_resp = self.session.get(self.url, timeout=10)
        original_length = len(original_resp.text)
        
        for category, payloads in self.PAYLOADS.items():
            for payload in payloads:
                # 构造测试URL
                test_params = params.copy()
                test_params[param] = [payload]
                test_query = urlencode(test_params, doseq=True)
                test_url = urlunparse((
                    parsed.scheme, parsed.netloc, parsed.path,
                    parsed.params, test_query, parsed.fragment
                ))
                
                try:
                    resp = self.session.get(test_url, timeout=15)
                    
                    # 检测错误信息
                    for pattern in self.ERROR_PATTERNS:
                        if re.search(pattern, resp.text, re.IGNORECASE):
                            return {
                                'type': 'error_based',
                                'param': param,
                                'payload': payload,
                                'evidence': pattern
                            }
                    
                    # 检测响应差异
                    if abs(len(resp.text) - original_length) > 500:
                        # 可能存在注入

    # XSS Payload

        if payload in response_text:
            return True
        
        # HTML解码后匹配
        import html
        decoded = html.unescape(response_text)
        if payload in decoded:
            return True
        
        # 部分匹配（检查关键部分）
        if '<script>' in payload and '<script>' in response_text.lower():
            return True
        if 'onerror=' in payload and 'onerror=' in response_text.lower():
            return True
        if 'onload=' in payload and 'onload=' in response_text.lower():
            return True
        
        return False
    
    def test_parameter(self, param: str, method: str = 'GET') -> Optional[Dict]:
        """测试单个参数"""

        params = parse_qs(parsed.query)

                    parsed.scheme, parsed.netloc, parsed.path,
                    parsed.params, test_query, parsed.fragment
                ))
                
                try:
                    resp = self.session.get(test_url, timeout=10)
                    if self.check_reflection(resp.text, payload):
                        return {
                            'type': 'reflected',
                            'method': 'GET',
                            'param': param,
                            'payload': payload,
                            'url': test_url
                        }

                    pass

                    resp = self.session.post(self.url, data=data, timeout=10)

                        return {
                            'type': 'reflected',
                            'method': 'POST',
                            'param': param,
                            'payload': payload
                        }

                    pass

                print(f"    类型: {result['type']}")
                print(f"    参数: {result['param']}")
                print(f"    Payload: {result['payload']}")
        
        return self.vulnerabilities

#!/usr/bin/env python3
"""
批量漏洞扫描脚本
整合多种漏洞检测
"""

import requests
import concurrent.futures
from typing import List, Dict
import argparse
import json
from datetime import datetime

class BatchScanner:
    def __init__(self, targets: List[str], threads: int = 10):
        self.targets = targets
        self.threads = threads
        self.results: Dict[str, Dict] = {}
    
    def scan_target(self, url: str) -> Dict:
        """扫描单个目标"""
        result = {
            'url': url,
            'timestamp': datetime.now().isoformat(),
            'vulnerabilities': []
        }
        
        try:
            # 基础检测
            resp = requests.get(url, timeout=10)
            
            # 检测安全头
            security_headers = {
                'X-Frame-Options': resp.headers.get('X-Frame-Options'),
                'X-XSS-Protection': resp.headers.get('X-XSS-Protection'),
                'X-Content-Type-Options': resp.headers.get('X-Content-Type-Options'),
                'Strict-Transport-Security': resp.headers.get('Strict-Transport-Security'),
                'Content-Security-Policy': resp.headers.get('Content-Security-Policy'),
            }
            
            missing_headers = [k for k, v in security_headers.items() if v is None]
            if missing_headers:
                result['vulnerabilities'].append({
                    'type': 'missing_security_headers',
                    'details': missing_headers
                })
            
            # 检测敏感信息泄露
            sensitive_patterns = [
                ('password', 'password exposure'),
                ('api_key', 'API key exposure'),
                ('secret', 'secret exposure'),
                ('token', 'token exposure'),
            ]
            
            for pattern, desc in sensitive_patterns:
                if pattern in resp.text.lower():
                    result['vulnerabilities'].append({
                        'type': 'information_disclosure',
                        'details': desc
                    })
                    break
            
            # 检测错误信息泄露
            error_patterns = [
                'stack trace',
                'exception',
                'error in',
                'warning:',
            ]
            
            for pattern in error_patterns:
                if pattern in resp.text.lower():
                    result['vulnerabilities'].append({
                        'type': 'error_disclosure',
                        'details': f'Found: {pattern}'
                    })
                    break
            
        except requests.RequestException as e:
            result['error'] = str(e)
        
        return result
    
    def run(self) -> Dict[str, Dict]:
        """执行批量扫描"""
        print(f"[*] 开始批量扫描 {len(self.targets)} 个目标")
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(self.scan_target, url): url for url in self.targets}
            
            completed = 0
            for future in concurrent.futures.as_completed(futures):
                url = futures[future]
                result = future.result()
                self.results[url] = result
                completed += 1
                
                vuln_count = len(result.get('vulnerabilities', []))
                print(f"[{completed}/{len(self.targets)}] {url} - {vuln_count} vulnerabilities")
        
        return self.results
    
    def save_results(self, filename: str):
        """保存结果"""
        with open(filename, 'w') as f:
            json.dump(self.results, f, indent=2, ensure_ascii=False)
        print(f"[+] 结果已保存: {filename}")
    
    def generate_report(self):
        """生成报告摘要"""
        total_vulns = sum(len(r.get('vulnerabilities', [])) for r in self.results.values())
        
        print("\n" + "="*50)
        print("扫描报告摘要")
        print("="*50)
        print(f"扫描目标数: {len(self.targets)}")
        print(f"发现漏洞数: {total_vulns}")
        
        # 按漏洞类型统计
        vuln_types = {}
        for result in self.results.values():
            for vuln in result.get('vulnerabilities', []):
                vuln_type = vuln['type']
                vuln_types[vuln_type] = vuln_types.get(vuln_type, 0) + 1
        
        print("\n漏洞类型分布:")
        for vuln_type, count in sorted(vuln_types.items()):
            print(f"  {vuln_type}: {count}")

def main():
    parser = argparse.ArgumentParser(description='批量漏洞扫描工具')
    parser.add_argument('-f', '--file', required=True, help='目标URL列表文件')
    parser.add_argument('-t', '--threads', type=int, default=10, help='线程数')
    parser.add_argument('-o', '--output', help='输出文件')
    args = parser.parse_args()
    
    # 读取目标列表
    with open(args.file, 'r') as f:
        targets = [line.strip() for line in f if line.strip()]
    
    scanner = BatchScanner(targets, args.threads)
    results = scanner.run()
    
    scanner.generate_report()
    
    if args.output:
        scanner.save_results(args.output)

if __name__ == '__main__':
    main()

        print(f"[+] 加载插件: {plugin.name}")
    
    def load_plugins_from_dir(self, plugin_dir: str) -> None:
        """从目录加载插件"""
        for filename in os.listdir(plugin_dir):
            if filename.endswith('.py') and not filename.startswith('_'):
                module_name = filename[:-3]
                module_path = os.path.join(plugin_dir, filename)
                
                try:
                    spec = importlib.util.spec_from_file_location(module_name, module_path)
                    module = importlib.util.module_from_spec(spec)
                    spec.loader.exec_module(module)
                    
                    for name, obj in inspect.getmembers(module):
                        if inspect.isclass(obj) and issubclass(obj, PluginBase) and obj != PluginBase:
                            self.load_plugin(obj())
                except Exception as e:
                    print(f"[-] 加载插件失败 {filename}: {e}")
    
    def scan(self, targets: List[str], plugins: List[str] = None) -> List[ScanResult]:
        """执行扫描"""
        selected_plugins = plugins or list(self.plugins.keys())
        
        print(f"[*] 开始扫描")
        print(f"[*] 目标数: {len(targets)}")
        print(f"[*] 插件数: {len(selected_plugins)}")
        
        def scan_target(target: str) -> List[ScanResult]:
            results = []
            for plugin_name in selected_plugins:
                if plugin_name in self.plugins:
                    plugin = self.plugins[plugin_name]
                    try:
                        plugin_results = plugin.scan(target)
                        for result in plugin_results:
                            result.timestamp = datetime.now().isoformat()
                            results.append(result)
                    except Exception as e:
                        print(f"[-] 插件 {plugin_name} 扫描失败: {e}")
            return results
        
        with concurrent.futures.ThreadPoolExecutor(max_workers=self.threads) as executor:
            futures = {executor.submit(scan_target, target): target for target in targets}
            
            for future in concurrent.futures.as_completed(futures):
                target = futures[future]
                try:
                    results = future.result()
                    self.results.extend(results)
                    print(f"[+] {target}: 发现 {len(results)} 个漏洞")
                except Exception as e:
                    print(f"[-] {target}: 扫描失败 {e}")
        
        return self.results
    
    def save_results(self, filename: str, format: str = 'json') -> None:
        """保存结果"""
        if format == 'json':
            data = [
                {
                    'target': r.target,
                    'plugin': r.plugin,
                    'vulnerability': r.vulnerability,
                    'severity': r.severity,
                    'description': r.description,
                    'evidence': r.evidence,
                    'timestamp': r.timestamp
                }
                for r in self.results
            ]
            with open(filename, 'w') as f:
                json.dump(data, f, indent=2, ensure_ascii=False)
        
        print(f"[+] 结果已保存: {filename}")
    
    def generate_report(self) -> None:
        """生成报告"""
        severity_count = {}
        for result in self.results:
            severity_count[result.severity] = severity_count.get(result.severity, 0) + 1
        
        print("\n" + "="*50)
        print("扫描报告")
        print("="*50)
        print(f"总漏洞数: {len(self.results)}")
        print("\n按严重程度:")
        for severity in ['critical', 'high', 'medium', 'low', 'info']:
            count = severity_count.get(severity, 0)
            if count > 0:
                print(f"  {severity}: {count}")

#!/usr/bin/env python3
"""
并发处理示例
多线程 vs 协程
"""

import time
import asyncio
import aiohttp
from concurrent.futures import ThreadPoolExecutor
from typing import List, Callable

# 多线程示例
def thread_worker(url: str) -> dict:
    """线程工作函数"""
    import requests
    try:
        resp = requests.get(url, timeout=5)
        return {'url': url, 'status': resp.status_code}
    except Exception as e:
        return {'url': url, 'error': str(e)}

def thread_scan(urls: List[str], workers: int = 10) -> List[dict]:
    """多线程扫描"""
    results = []
    with ThreadPoolExecutor(max_workers=workers) as executor:
        futures = [executor.submit(thread_worker, url) for url in urls]
        for future in futures:
            results.append(future.result())
    return results

# 协程示例
async def async_worker(session: aiohttp.ClientSession, url: str) -> dict:
    """协程工作函数"""
    try:
        async with session.get(url, timeout=aiohttp.ClientTimeout(total=5)) as resp:
            return {'url': url, 'status': resp.status}
    except Exception as e:
        return {'url': url, 'error': str(e)}

async def async_scan(urls: List[str], workers: int = 10) -> List[dict]:
    """协程扫描"""
    connector = aiohttp.TCPConnector(limit=workers)
    async with aiohttp.ClientSession(connector=connector) as session:
        tasks = [async_worker(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
    return list(results)

# 性能对比
def benchmark():
    """性能测试"""
    urls = ['http://httpbin.org/delay/1'] * 20
    
    # 多线程
    start = time.time()
    thread_results = thread_scan(urls, workers=10)
    thread_time = time.time() - start
    print(f"多线程: {thread_time:.2f}s")
    
    # 协程
    start = time.time()
    async_results = asyncio.run(async_scan(urls, workers=10))
    async_time = time.time() - start
    print(f"协程: {async_time:.2f}s")

package main

import (
    "fmt"
    "net"
    "sync"
    "flag"
    "time"
)

func scanPort(host string, port int, timeout time.Duration) bool {
    target := fmt.Sprintf("%s:%d", host, port)
    conn, err := net.DialTimeout("tcp", target, timeout)
    if err != nil {
        return false
    }
    conn.Close()
    return true
}

func worker(host string, ports <-chan int, results chan<- int, timeout time.Duration, wg *sync.WaitGroup) {
    defer wg.Done()
    for port := range ports {
        if scanPort(host, port, timeout) {
            results <- port
        }
    }
}

func main() {
    host := flag.String("host", "127.0.0.1", "目标主机")
    startPort := flag.Int("start", 1, "起始端口")
    endPort := flag.Int("end", 1024, "结束端口")
    workers := flag.Int("workers", 100, "工作协程数")
    timeout := flag.Duration("timeout", 1*time.Second, "超时时间")
    flag.Parse()

    ports := make(chan int, *workers)
    results := make(chan int)
    var wg sync.WaitGroup

    // 启动工作协程
    for i := 0; i < *workers; i++ {
        wg.Add(1)
        go worker(*host, ports, results, *timeout, &wg)
    }

    // 收集结果
    go func() {
        wg.Wait()
        close(results)
    }()

    // 发送端口
    go func() {
        for port := *startPort; port <= *endPort; port++ {
            ports <- port
        }
        close(ports)
    }()

    // 输出结果
    for port := range results {
        fmt.Printf("[+] %s:%d open\n", *host, port)
    }
}

package main

import (
    "fmt"
    "io"
    "net/http"
    "sync"
    "time"
    "strings"
)

type Result struct {
    URL        string
    StatusCode int
    Length     int
    Title      string
}

func scanURL(client *http.Client, url string) (*Result, error) {
    resp, err := client.Get(url)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    body, _ := io.ReadAll(resp.Body)
    title := extractTitle(string(body))

    return &Result{
        URL:        url,
        StatusCode: resp.StatusCode,
        Length:     len(body),
        Title:      title,
    }, nil
}

func extractTitle(html string) string {
    start := strings.Index(html, "<title>")
    end := strings.Index(html, "</title>")
    if start != -1 && end != -1 && end > start {
        return html[start+7 : end]
    }
    return ""
}

func main() {
    urls := []string{
        "https://example.com",
        "https://google.com",
        "https://github.com",
    }

    client := &http.Client{
        Timeout: 10 * time.Second,
    }

    var wg sync.WaitGroup
    results := make(chan *Result, len(urls))

    for _, url := range urls {
        wg.Add(1)
        go func(u string) {
            defer wg.Done()
            result, err := scanURL(client, u)
            if err == nil {
                results <- result
            }
        }(url)
    }

    go func() {
        wg.Wait()
        close(results)
    }()

    for result := range results {
        fmt.Printf("[%d] %s - %s (%d bytes)\n",
            result.StatusCode, result.URL, result.Title, result.Length)
    }
}
